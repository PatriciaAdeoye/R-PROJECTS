### 7. Model Performance Evaluation
# Predictions
lm_pred <- predict(model_lm, data)
lasso_pred <- predict(cv_lasso, newx = x_matrix, s = "lambda.min")
tree_pred <- predict(tree_model, data)
rf_pred <- predict(rf_model, data)

# Compute RMSE and R
metrics <- function(true, pred) {
  data.frame(
    RMSE = sqrt(mean((true - pred)^2)),
    R2 = cor(true, pred)^2
  )
}

results <- bind_rows(
  metrics(y, lm_pred) %>% mutate(Model = "Linear Regression"),
  metrics(y, lasso_pred) %>% mutate(Model = "Lasso Regression"),
  metrics(y, tree_pred) %>% mutate(Model = "Regression Tree"),
  metrics(y, rf_pred) %>% mutate(Model = "Random Forest")
)

print(results)
